<?php

namespace Charcoal\Cms\Section;

use \Charcoal\Cms\AbstractSection;
use Charcoal\Cms\EmptySectionInterface;

/**
 * Empty section
 *
 * Empty sections are linked to a static template and do not provide any dynamic content
 * (except standard section metadata).
 */
class EmptySection extends AbstractSection implements
    EmptySectionInterface
{
    /**
     * @return string
     */
    public function sectionType()
    {
        return AbstractSection::TYPE_EMPTY;
    }
}
