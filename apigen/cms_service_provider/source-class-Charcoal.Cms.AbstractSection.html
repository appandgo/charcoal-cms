457:     /**
458:      * Parse the property value as a "multiple" value type.
459:      *
460:      * @param  mixed                    $value     The value being converted to an array.
461:      * @param  string|PropertyInterface $separator The boundary string.
462:      * @return array
463:      */
464:     public function parseAsMultiple($value, $separator = ',')
465:     {
466:         if (!isset($value) ||
467:             (is_string($value) && !strlen(trim($value))) ||
468:             (is_array($value) && !count(array_filter($value, 'strlen')))
469:         ) {
470:             return [];
471:         }
472: 
473:         /**
474:          * This property is marked as "multiple".
475:          * Manually handling the resolution to array
476:          * until the property itself manages this.
477:          */
478:         if (is_string($value)) {
479:             return explode($separator, $value);
480:         }
481: 
482:         /**
483:          * If the parameter isn't an array yet,
484:          * means we might be dealing with an integer,
485:          * an empty string, or an object.
486:          */
487:         if (!is_array($value)) {
488:             return [ $value ];
489:         }
490: 
491:         return $value;
492:     }
